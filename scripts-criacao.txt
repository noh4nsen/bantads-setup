CREATE DATABASE Autenticacao;
CREATE TABLE Usuario (
	id SERIAL Primary Key,
	email varchar(100),
	senha varchar(400),
	tipoUsuario int
)
--------------------------
CREATE DATABASE Gerente;
CREATE TABLE Gerente (
	id SERIAL Primary Key,
	nome varchar(60),
	cpf varchar(11) unique,
	idUsuario int
)	
--------------------------
CREATE DATABASE Cliente;
CREATE TABLE Edereco(
	id SERIAL Primary Key,
	cep varchar(8),
	logradouro varchar(100),
	numero int,
	complemento varchar(100),
	bairro varchar(60),
	cidade varchar(60),
	estado barchar(60)
) 
CREATE TABLE Analise(
	id SERIAL Primary Key,
	aprovado boolean,
	motivo varchar(400),
	datahora date,
	idGerente int
)
CREATE TABLE Cliente (
	id SERIAL Primary Key,
	cpf varchar(11) unique,
	salario decimal(9,2),
	idUsuario int,
	idEndereco int,
	idAnalise int,
	CONSTRAINT fk_endereco FOREIGN KEY(idEndereco) REFERENCES Endereco(id),
	CONSTRAINT fk_analise FOREIGN KEY(idAnalise) REFERENCES Analise(id)
)
-------------------------
CREATE DATABASE ContaCUD;
CREATE TABLE Conta(
	id SERIAL Primary Key,
	numero serial,
	saldo decimal(9,2),
	datacriacao date,
	limite decimal(9,2),
	idCliente int
)
CREATE TABLE Movimentacao (
	id SERIAL Primary Key,
	valor decimal(9,2),
	datahora date,
	tipomovimentacao int,
	origem int,
	destino int,
	CONSTRAINT fk_contaorigem FOREIGN KEY(origem) REFERENCES Conta(id),
	CONSTRAINT fk_contadestino FOREIGN KEY(destino) REFERENCES Conta(id)
)
